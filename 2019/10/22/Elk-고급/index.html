/usr/lib/jvm/java-1.8.0-openjdk-1.8.0.222.b10-1.el7_7.x86_64/bin/javac

1 확인

root@zetawiki:~# echo $JAVA_HOME

root@zetawiki:~# javac -version
javac 1.7.0_91

2 javac 위치 확인

root@zetawiki:~# which javac
/usr/bin/javac

root@zetawiki:~# readlink -f /usr/bin/javac
/usr/lib/jvm/java-7-openjdk-amd64/bin/javac

→ javac의 실제 위치는 /usr/lib/jvm/java-7-openjdk-amd64/bin/

→ 따라서 $JAVA_HOME 은 /usr/lib/jvm/java-7-openjdk-amd64 으로 설정해야 함

3 JAVA_HOME

  • /etc/profile 하단에 내용 추가

root@zetawiki:~# vi /etc/profile

export JAVA_HOME=/usr/lib/jvm/java-7-openjdk-amd64

4 확인

  • SSH 재접속 후 echo로 확인

root@zetawiki:~# echo $JAVA_HOME
/usr/lib/jvm/java-7-openjdk-amd64

root@zetawiki:~# $JAVA_HOME/bin/javac -version
javac 1.7.0_91
